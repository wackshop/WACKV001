        local corner = Instance.new("UICorner", button)
        corner.CornerRadius = UDim.new(0, 8)

        button.MouseButton1Click:Connect(function()

                return game:HttpGet(url, true)
            end)
            if success then
                local loadSuccess, loadResult = pcall(function()
                    loadstring(result)()
